ACTOR Rail_Standard
{
	Radius				11
	Height				8
	Speed				30
	Damage				1
	DamageType			MarineFire
	
	DamageType			MarineFire
	RenderStyle			Add
	
	Projectile
	+RANDOMIZE
	+DEHEXPLOSION
	
	explosiondamage			90	// default, 128
	explosionradius			90	// default, 128
	
	SeeSound			"x1/fire1"
	//DeathSound			"weapons/rocklx"
	Obituary			"%o smells %k's rails... oh wait, thats $o's melted insides"
        DontHurtShooter
	
	States
	{
		Spawn:
			PLSS A 1 bright A_SpawnItem ("Rail_Standard_Tracer" , 0, 0, 0, 0)
			PLSS B 1
			goto Spawn
		Death:
			PLSE C 8 bright A_Explode
			PLSE D 6 bright
			PLSE E 4 bright
			stop
	}
}
