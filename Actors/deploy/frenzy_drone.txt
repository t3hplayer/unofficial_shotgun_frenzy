ACTOR sfDroneSaw
{
	hitobituary "%o was shaved by a Saw Drone"
	health 5
	painchance 180
	speed 8
	radius 10
	height 10
	mass 400
	seesound "p/drone_saw1"
	attacksound "p/drone_saw3"
	
	scale 0.7
	explosionradius 32
	explosiondamage 8
	
	MONSTER
	+FRIENDLY
	+FLOORCLIP
	+NEVERRESPAWN
	states
	{
		Spawn:
			SAWD AB 1 A_Look
			goto Spawn
		See:
			SAWD A 0 A_PlaySound("p/drone_saw2")
			SAWD AABB 1 A_Chase
			goto See
		Melee:
			SAWD AB 1 A_FaceTarget
			SAWD B 1 A_SargAttack
			goto See
		Pain:
			SAWD A 1
			SAWD B 1 A_Pain
			goto See
		Death:
			MISL B 8 bright A_Explode
			MISL C 6 bright
			MISL D 4 bright
			stop
		Raise:
			SAWD NMLKJI 5
			goto See
	}
}
