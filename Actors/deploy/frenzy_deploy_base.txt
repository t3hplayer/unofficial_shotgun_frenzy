//Base and dummy actors go here

//Construction check
actor build_check : inventory
{
	inventory.amount 1
	inventory.maxamount 1
}

//*_Building actors base
actor BuildBase
{
	states
	{
		Spawn:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInTargetInventory("build_check", 1, 2)
		TNT1 A 0 A_Jump(256, "Spawn2")
		TNT1 A 1
		goto Spawn
	}
}
